function ghDownload() { if(!ghToken) { alert("请先填入 GitHub Token"); return; } setGhSyncSt("syncing"); fetch(GH_API,{headers:{Authorization:"token "+ghToken.trim(),Accept:"application/vnd.github.v3+json"}}).then(function(r) { if(!r.ok) { return r.text().then(function(t) { throw new Error("HTTP "+r.status+": "+t.slice(0,200)); }); } return r.json(); }).then(function(d) { if(!d.content) throw new Error("文件内容为空"); var payload=JSON.parse(decodeURIComponent(escape(atob(d.content.replace(/\n/g,""))))); if(payload.accts) setAccts(payload.accts); if(payload.txns) setTxns(payload.txns); if(payload.rates) setRates(payload.rates); if(payload.base) setBase(payload.base); if(payload.vis!==undefined) setVis(payload.vis); if(payload.acctOrder) setAcctOrder(payload.acctOrder); var now=new Date().toLocaleString("zh-CN",{month:"numeric",day:"numeric",hour:"2-digit",minute:"2-digit"}); setLastGhSync(now); localStorage.setItem("cb_last_gh_sync",now); setGhSyncSt("ok"); }).catch(function(e) { alert("下载失败: "+e.message); setGhSyncSt("err"); }); setTimeout(function() { setGhSyncSt(""); },5000); } Cashbook
